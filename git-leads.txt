git checkout -- <имя файла> - откатывает изменения в файле до предыдущего коммита.

ssh-keygen - создает пару SSH ключей
vi <путь>/id_rsa.pub - показывает публичный SSH ключ

git add --all = git add .

git commit -a -m "<текст коммита>" - "-a" коммитит те файлы, которые уже в стейджинге.

git branch -d <имя ветки>

git reset —hard отменяет последний коммит, удаляет изменения из стейджинга и удаляет все изменения до предыдущего коммита
git reset —mixed отменяет последний коммит и удаляет изменения из стейджинга (вариант по умолчанию)
git reset —soft отменяет последний коммит, но оставляет изменения в стейджинге.
